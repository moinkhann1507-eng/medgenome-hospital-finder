   655→   655→// Levenshtein distance for fuzzy matching
   656→   656→function levenshtein(a, b) {
   657→   657→  const m = a.length, n = b.length;
   658→   658→  const dp = Array.from({length: m+1}, () => Array(n+1).fill(0));
   659→   659→  for (let i = 0; i <= m; i++) dp[i][0] = i;
   660→   660→  for (let j = 0; j <= n; j++) dp[0][j] = j;
   661→   661→  for (let i = 1; i <= m; i++) {
   662→   662→    for (let j = 1; j <= n; j++) {
   663→   663→      dp[i][j] = Math.min(
   664→   664→        dp[i-1][j] + 1,
   665→   665→        dp[i][j-1] + 1,
   666→   666→        dp[i-1][j-1] + (a[i-1] !== b[j-1] ? 1 : 0)
   667→   667→      );
   668→   668→    }
   669→   669→  }
   670→   670→  return dp[m][n];
   671→   671→}

   673→   673→// Fuzzy match score (0-1, 1 = perfect)
   674→   674→function fuzzyScore(query, target) {
   675→   675→  const q = query.toLowerCase();
   676→   676→  const t = target.toLowerCase();
   677→   677→  if (t.includes(q)) return 1;
   678→   678→  if (q.split(' ').every(w => t.includes(w))) return 0.95;
   679→   679→  const dist = levenshtein(q, t);
   680→   680→  const maxLen = Math.max(q.length, t.length);
   681→   681→  return Math.max(0, 1 - dist / maxLen);
   682→   682→}

   753→   753→function scoreHospital(hospital, parsed) {
   754→   754→  let score = 0;
   755→   755→
   756→   756→  if (parsed.detectedName) {
   757→   757→    score += fuzzyScore(parsed.detectedName, hospital.name) * 100;
   758→   758→    return score; // Name match is definitive

   762→   762→  if (parsed.detectedCity && hospital.city.toLowerCase() === parsed.detectedCity.toLowerCase()) score += 35;
   763→   763→  if (parsed.detectedCountry && hospital.country.toLowerCase() === parsed.detectedCountry.toLowerCase()) score += 25;
   764→   764→  if (parsed.isEmergency && hospital.emergency) score += 20;
   765→   765→
   766→   766→  // Fuzzy name matching against query terms
   767→   767→  const terms = document.getElementById('searchInput').value.toLowerCase().split(/\s+/).filter(w => w.length > 2);
   768→   768→  for (const term of terms) {
   769→   769→    if (['find','hospital','near','in','the','show','best','top','any','all','get','locate','search','for','me','my'].includes(term)) continue;
   770→   770→    score += fuzzyScore(term, hospital.name) * 8;
   771→   771→    score += fuzzyScore(term, hospital.city) * 6;
   772→   772→    score += fuzzyScore(term, hospital.specialty.toLowerCase()) * 5;
   773→   773→    score += fuzzyScore(term, hospital.type) * 4;
   774→   774→  }
   775→   775→
   776→   776→  // Bonus for rating
   777→   777→  score += hospital.rating * 2;
   778→   778→
   779→   779→  return score;
   780→   780→}

  1358→  1358→// Haversine distance in km
  1359→  1359→function getDistance(lat1, lon1, lat2, lon2) {
  1360→  1360→  const R = 6371;
  1361→  1361→  const dLat = (lat2 - lat1) * Math.PI / 180;
  1362→  1362→  const dLon = (lon2 - lon1) * Math.PI / 180;
  1363→  1363→  const a = Math.sin(dLat/2) * Math.sin(dLat/2) +
  1364→  1364→    Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
  1365→  1365→    Math.sin(dLon/2) * Math.sin(dLon/2);
  1366→  1366→  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
  1367→  1367→}
